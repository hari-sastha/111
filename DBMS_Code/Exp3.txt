-- 1. Initialize the Database
CREATE DATABASE IF NOT EXISTS LibraryManagementSystem;
USE LibraryManagementSystem;

-- 2. Create Schema in Relational Order
-- Author must exist before Book
CREATE TABLE Author (
    AuthorID INT PRIMARY KEY,
    AuthorName VARCHAR(255),
    BirthDate DATE
);

-- Book must exist before Transactions
CREATE TABLE Book (
    ISBN VARCHAR(20) PRIMARY KEY,
    Title VARCHAR(255),
    AuthorID INT,
    Genre VARCHAR(50),
    FOREIGN KEY (AuthorID) REFERENCES Author(AuthorID)
);

-- Member must exist before Transactions
CREATE TABLE Member (
    MemberID INT PRIMARY KEY,
    MemberName VARCHAR(255),
    Email VARCHAR(255)
);

-- Transactions links Books and Members
CREATE TABLE Transactions (
    TransactionID INT PRIMARY KEY,
    BookID VARCHAR(20),
    MemberID INT,
    CheckoutDate DATE,
    ReturnDate DATE,
    FOREIGN KEY (BookID) REFERENCES Book(ISBN),
    FOREIGN KEY (MemberID) REFERENCES Member(MemberID)
);
