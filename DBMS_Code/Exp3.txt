CREATE TABLE Books (
    BookID INTEGER PRIMARY KEY,
    Title TEXT
);

CREATE TABLE Members (
    MemberID INTEGER PRIMARY KEY,
    Name TEXT
);

CREATE TABLE Borrow (
    BorrowID INTEGER PRIMARY KEY,
    BookID INTEGER,
    MemberID INTEGER,
    IssueDate TEXT,
    ReturnDate TEXT,
    FOREIGN KEY (BookID) REFERENCES Books(BookID),
    FOREIGN KEY (MemberID) REFERENCES Members(MemberID)
);
